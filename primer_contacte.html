#include <stdio.h>
#define PI 3.14159

int main() {
  float radi, area;
  printf("Càlcul de l'àrea d'un cercle\n");
  printf("Introduïu el radi: ");
  scanf("%f", &radi);               /* conversió de String a float */
  area = PI * radi * radi;
  printf("L'àrea és %f\n", area);
  return 0;
}

import java.io.*;

public class CalculaRadi {
  public static void main(String[] args) throws IOException {
    System.out.println("Càlcul de l'àrea d'un cercle");
    System.out.print("Introduïu el radi: ");
    BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
    String linia = reader.readLine();
    float radi = Float.parseFloat(linia);  // conversió de String a float
    float area =  (float)Math.PI * radi * radi;
    System.out.println("L'àrea és " + area);
  }
}

import java.io.*
fun main() {
    val reader = BufferedReader(InputStreamReader(System.`in`))
    println("Càlcul de l'àrea d'un cercle")
    print("Introduïu el radi: ")
    val linia = reader.readLine()
    val radi = linia.toFloat()      // conversió de String a float
    val area = Math.PI.toFloat() * radi * radi
    println("L'àrea és $area")
}
